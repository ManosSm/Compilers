def main_factorial():
#{
#$ declarations #$
#declare x
#declare i,fact
#$ body of main_factorial #$
x = int(input());
fact = 1;
i = 1;
while (i<=x):
#{
fact = fact * i;
i = i + 1;
#}
print(fact);
#}
def main_fibonacci():
#{
#declare x
def fibonacci(x):
#{
if (x<=1):
return(x);
else:
return (fibonacci(x-1)+fibonacci(x-2));
#}
x = int(input());
print(fibonacci(x));
#}
